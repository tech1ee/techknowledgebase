fun isHappy(n: Int): Boolean {
    var slow = n
    var fast = n

    do {
        slow = sumOfSquares(slow)
        fast = sumOfSquares(sumOfSquares(fast))
    } while (slow != fast)

private fun sumOfSquares(n: Int): Int {
    var num = n
    var sum = 0
    while (num > 0) {
        val digit = num % 10
        sum += digit * digit
        num /= 10
    }
    return sum
}

 *         –°—Ä–∞–≤–Ω–∏—Ç—å: 1‚Üí2 vs 1‚Üí2 ‚úì

fun isPalindrome(head: ListNode?): Boolean {
    if (head?.next == null) return true

    // 1. –ù–∞–π—Ç–∏ —Å–µ—Ä–µ–¥–∏–Ω—É
    var slow = head
    var fast = head
    while (fast?.next != null && fast.next?.next != null) {
        slow = slow?.next
        fast = fast.next?.next
    }

    // 2. –†–∞–∑–≤–µ—Ä–Ω—É—Ç—å –≤—Ç–æ—Ä—É—é –ø–æ–ª–æ–≤–∏–Ω—É
    var secondHalf = reverseList(slow?.next)

    // 3. –°—Ä–∞–≤–Ω–∏—Ç—å
    var firstHalf = head
    while (secondHalf != null) {
        if (firstHalf?.`val` != secondHalf.`val`) {
            return false
        }
        firstHalf = firstHalf.next
        secondHalf = secondHalf.next
    }

    return true
}

private fun reverseList(head: ListNode?): ListNode? {
    var prev: ListNode? = null
    var curr = head
    while (curr != null) {
        val next = curr.next
        curr.next = prev
        prev = curr
        curr = next
    }
    return prev
}
